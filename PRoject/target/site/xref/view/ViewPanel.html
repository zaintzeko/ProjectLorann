17  class ViewPanel extends JPanel implements Observer {
18  
19  	/** The Constant serialVersionUID. */
20  	private static final long	serialVersionUID	= -998294702363713521L;
21  	
22  	/** The view frame. */
23  	private ViewFrame					viewFrame;
24  
25  	/** The lorann world. */
26  	private final ILorannWorld lorannWorld;
27  
28  	/**
29  	 * Instantiates a new view panel.
30  	 *
31  	 * @param viewFrame the view frame
32  	 * @param lorannWorld the lorann world
33  	 */
34  	public ViewPanel(final ViewFrame viewFrame, final ILorannWorld lorannWorld) {
35  		this.setViewFrame(viewFrame);
36  		viewFrame.getModel().getObservable().addObserver(this);
37  		this.lorannWorld = lorannWorld;
38  	}
39  
40  	/**
41  	 * Gets the view frame.
42  	 *
43  	 * @return the view frame
44  	 */
45  	private ViewFrame getViewFrame() {
46  		return this.viewFrame;
47  	}
48  
49  	/*
50  	 * (non-Javadoc)
51  	 *
52  	 * @see javax.swing.JComponent#paintComponent(java.awt.Graphics)
53  	 */
54  	@Override
55  	protected void paintComponent(final Graphics graphics) {
56  		graphics.clearRect(0, 0, this.lorannWorld.getWidth(), this.lorannWorld.getHeight());
57  		for(int y = 0; y < 12; y++)
58  		{
59  			for(int x = 0; x < 20; x++)
60  			{
61  				if(this.lorannWorld.getElement(x, y)!=null) {
62  					//System.out.println(this.motionlessElements[x][y].getSymbole());
63  					graphics.drawImage(this.lorannWorld.getElement(x, y).getSprite().getImage(), x*32, y*32, null);
64  				} else {
65  					graphics.setColor(new Color(0,0,0));
66  					graphics.fillRect(x*32, y*32, 32, 32);
67  				}
68  			}
69  		}
70  		for(int i = 0; i < 20; i++)	{
71  			graphics.setColor(new Color(0,0,0));
72  			graphics.fillRect(i*32, 12*32, 32, 32);
73  		}
74  		for(final IMotionElement h : this.lorannWorld.getMotionElements())
75  		{
76  			graphics.drawImage(h.getSprite().getImage(), h.getX()*32, h.getY()*32, null);
77  		}
78  		graphics.drawImage(this.lorannWorld.getLorann().getSpell().getSprite().getImage(), this.lorannWorld.getLorann().getSpell().getX()*32, this.lorannWorld.getLorann().getSpell().getY()*32, null);
79  		graphics.drawImage(this.lorannWorld.getLorann().getSprite().getImage(), this.lorannWorld.getLorann().getX()*32, this.lorannWorld.getLorann().getY()*32, null);
80  		graphics.setColor(new Color(255,0,0));
81  		graphics.drawString("score : " + this.lorannWorld.getScore(), 0, (12*32)-10);
82  	}
83

85  	 * Sets the view frame.
86  	 *
87  	 * @param viewFrame the new view frame
88  	 */
89  	private void setViewFrame(final ViewFrame viewFrame) {
90  		this.viewFrame = viewFrame;
91  	}

97  	 */
98  
99  	public void update(final Observable o, final Object arg) {
100 		this.repaint();
101 	}
